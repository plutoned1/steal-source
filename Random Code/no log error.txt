[HarmonyPatch(typeof(GorillaNot), "LogErrorCount", MethodType.Normal)]
        public class NoLogError : MonoBehaviour
        {
            static bool Prefix(string logString, string stackTrace, LogType type)
            {
                //Debug.unityLogger.logEnabled = false;
                Debug.Log("Blocked Log LogString: " + logString + " StackTrace: " + stackTrace + " Log type: " + type.ToString());
                return false;
            }
        }
